    13	/**
    14	* Converts a vector of polynomial limbs to a single flattened array
    15	*/
    16	std::vector<std::vector<uint64_t>> FIDESlib::CKKS::GetRawArray(
    17	    std::vector<lbcrypto::PolyImpl<lbcrypto::NativeVector>> polys) {
    18	    // total size is r * N
    19	    int numRes = polys.size();
    20	    int numElements = (polys[0].GetValues() /*.m_values*/).GetLength();
    21	
    22	    std::vector<std::vector<uint64_t>> flattened(numRes, std::vector<uint64_t>(numElements));
    23	
    24	    // Fill the array
    25	    for (int r = 0; r < numRes; ++r) {
    26	        for (int i = 0; i < numElements; i++) {
    27	            flattened[r][i] = (polys[r].GetValues() /*.m_values*/)[i].ConvertToInt();
    28	        }
    29	    }
    30	    return flattened;
    31	};

    34	* Gets the moduli from a vector of polynomial limbs and returns a single array
    35	*/
    36	static std::vector<uint64_t> GetModuli(std::vector<lbcrypto::PolyImpl<lbcrypto::NativeVector>> polys) {
    37	    int numRes = polys.size();
    38	    std::vector<uint64_t> moduli(numRes);
    39	    for (int r = 0; r < numRes; r++) {
    40	        moduli[r] = polys[r].GetModulus().ConvertToInt();
    41	    }
    42	    return moduli;
    43	};
