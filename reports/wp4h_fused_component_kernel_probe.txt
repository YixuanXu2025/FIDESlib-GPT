   308	        INTT_<T, false, algo><<<gridDim, blockDim, bytes, stream.ptr()>>>(getGlobals(), v.data, primeid, aux.data);
   309	
   310	        blockDim = (1 << ((cc.logN + 1) / 2 - 1));
   311	        gridDim = {v.size / blockDim.x / 2 / M};
   312	        bytes = sizeof(T) * blockDim.x * (2 * M + 1 + (algo == 2 || algo == 3 ? 1 : 0));
   313	
   314	        INTT_<T, true, algo><<<gridDim, blockDim, bytes, stream.ptr()>>>(getGlobals(), aux.data, primeid, v.data);
   315	    }
   316	}

   318	template <typename T>
   319	template <ALGO algo>
   320	void Limb<T>::NTT() {
   321	
   322	    if constexpr (0) {
   323	        assert(primeid >= 0);
   324	        constexpr int M = sizeof(T) == 8 ? 4 : 8;
   325	        constexpr int K = 1;
   326	
   327	        dim3 blockDim{(uint32_t)(1 << ((cc.logN + 1) / 2 - 1)), K};
   328	        dim3 gridDim{v.size / blockDim.x / 2 / M};
   329	        int bytes = sizeof(T) * blockDim.x * (2 * M + 1 + (algo == 2 || algo == 3 ? 1 : 0));
   330	
   331	        //T *psi_arr = (T *) G_::psi[primeid];
   332	        //T *psi_arr_middle_scale = (T *) G_::psi_middle_scale[primeid];
   333	
   334	        int primeid_ = primeid;
   335	        void* args[6] = {getGlobals(), &v.data, (void*)&primeid_, &aux.data, (void*)&primeid_, (void*)&primeid_};
   336	        cudaLaunchCooperativeKernel(get_NTT_reference(false) /*(void *) test_kernel*/ /*(void *) NTT_<T, false, algo>*/,
   337	                                    gridDim, blockDim, args, bytes, stream.ptr());
   338	        //CudaCheckErrorModNoSync;
   339	    } else if constexpr (1) {
   340	        static std::map<int, cudaGraphExec_t> exec;
   341	
   342	        run_in_graph<false>(exec[primeid], stream, [&]() {
   343	            assert(primeid >= 0);
   344	            constexpr int M = sizeof(T) == 8 ? 4 : 8;
   345	
   346	            dim3 blockDim{(uint32_t)(1 << ((cc.logN + 1 + (cc.logN > 13 ? 0 : 0)) / 2 - 1))};
   347	            dim3 gridDim{v.size / blockDim.x / 2 / M};
   348	            int bytes = sizeof(T) * blockDim.x * (2 * M + 1 + (algo == 2 || algo == 3 ? 1 : 0));
   349	
   350	            NTT_<T, false, algo><<<gridDim, blockDim, bytes, stream.ptr()>>>(getGlobals(), v.data, primeid, aux.data);
   351	
   352	            {
   353	                blockDim = dim3{(uint32_t)(1 << ((cc.logN + (cc.logN > 13 ? 0 : 0)) / 2 - 1))};
   354	                gridDim = {v.size / blockDim.x / 2 / M};
   355	                bytes = sizeof(T) * blockDim.x * (2 * M + 1 + (algo == 2 || algo == 3 ? 1 : 0));
   356	
   357	                NTT_<T, true, algo>
   358	                    <<<gridDim, blockDim, bytes, stream.ptr()>>>(getGlobals(), aux.data, primeid, v.data);
   359	            }
   360	        });

   214	void LimbPartition::generate(std::vector<LimbRecord>& records, std::vector<LimbImpl>& limbs, VectorGPU<void*>& ptrs,
   215	                             int pos, VectorGPU<void*>* auxptrs, uint64_t* buffer, size_t offset, uint64_t* buffer_aux,
   216	                             size_t offset_aux, bool noptr) {
   217	    CudaNvtxRange r(std::string{sc::current().function_name()}.substr());
   218	    constexpr bool USE_PARTITION_STREAM = true;
   219	    assert(pos < (int)records.size());
   220	    cudaSetDevice(device);
   221	
   222	    const int limbs_size = limbs.size();
   223	    int size = std::max((int)(pos - limbs_size + 1), (int)0);
   224	    std::vector<void*> cpu_ptr(size, nullptr);
   225	    std::vector<void*> cpu_auxptr(size, nullptr);
   226	    for (int i = limbs_size; i <= pos; ++i) {
   227	        const LimbRecord& r = records.at(i);
   228	        if (r.type == U32) {
   229	            if (buffer && buffer_aux) {
   230	                limbs.emplace_back(Limb<uint32_t>(cc, (uint32_t*)buffer, 2 * offset, id,
   231	                                                  USE_PARTITION_STREAM ? s : records.at(i).stream, r.id,
   232	                                                  (uint32_t*)buffer_aux, 2 * offset_aux));
   233	                offset += cc.N;
   234	                offset_aux += cc.N;
   235	            } else if (buffer) {
   236	                limbs.emplace_back(Limb<uint32_t>(cc, (uint32_t*)buffer, 2 * offset, id,
   237	                                                  USE_PARTITION_STREAM ? s : records.at(i).stream, r.id, nullptr, 0));
   238	                offset += cc.N;
   239	            } else
   240	                limbs.emplace_back(
   241	                    Limb<uint32_t>(cc, id, USE_PARTITION_STREAM ? s : records.at(i).stream, r.id, auxptrs ? 1 : 0));
   242	            cpu_ptr[i - limbs_size] = {&(std::get<U32>(limbs.back()).v.data)[0]};
   243	            cpu_auxptr[i - limbs_size] = {&(std::get<U32>(limbs.back()).aux.data)[0]};
   244	        }
   245	        if (r.type == U64) {
   246	            if (buffer && buffer_aux) {
   247	                limbs.emplace_back(Limb<uint64_t>(cc, buffer, offset, id,
   248	                                                  USE_PARTITION_STREAM ? s : records.at(i).stream, r.id, buffer_aux,
   249	                                                  offset_aux));
   250	                offset += cc.N;
   251	                offset_aux += cc.N;
   252	            } else if (buffer) {
   253	                limbs.emplace_back(Limb<uint64_t>(cc, buffer, offset, id,
   254	                                                  USE_PARTITION_STREAM ? s : records.at(i).stream, r.id, nullptr, 0));
   255	                offset += cc.N;
   256	            } else {
   257	                if (auxptrs) {
   258	                    limbs.emplace_back(
   259	                        Limb<uint64_t>(cc, id, USE_PARTITION_STREAM ? s : records.at(i).stream, r.id, false));
   260	                } else {
   261	                    limbs.emplace_back(
   262	                        Limb<uint64_t>(cc, id, USE_PARTITION_STREAM ? s : records.at(i).stream, r.id, true));
   263	                }
   264	            }
   265	
   266	            cpu_ptr[i - limbs_size] = {&(std::get<U64>(limbs.back()).v.data)[0]};
   267	            cpu_auxptr[i - limbs_size] = {&(std::get<U64>(limbs.back()).aux.data)[0]};
   268	            //cudaFreeHost(aux);
   269	        }
   270	        CudaCheckErrorModNoSync;
   271	        s.wait(USE_PARTITION_STREAM ? s : records.at(i).stream);
   272	        CudaCheckErrorModNoSync;
   273	    }
   274	    if (size > 0) {
   275	        if (!noptr) {
   276	            cudaMemcpyAsync(ptrs.data + limbs_size, cpu_ptr.data(), size * sizeof(void*), cudaMemcpyHostToDevice,
   277	                            s.ptr());
   278	            CudaCheckErrorModNoSync;
   279	            if (auxptrs) {
   280	                cudaMemcpyAsync((*auxptrs).data + limbs_size, cpu_auxptr.data(), size * sizeof(void*),
   281	                                cudaMemcpyHostToDevice, s.ptr());
   282	            }
   283	            CudaCheckErrorModNoSync;
   284	        }
   285	    }
   286	    CudaCheckErrorModNoSync;
   287	}
